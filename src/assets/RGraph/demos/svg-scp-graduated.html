A graduated SVG SemiCircular Progress chart

If you like RGraph please share it:

This goes in the documents header: 
<script src="RGraph.svg.common.core.js"></script>
<script src="RGraph.svg.semicircularprogress.js"></script>
Put this where you want the chart to show up: 
<div style="width: 450px; height: 250px" id="chart-container"></div>
This is the code that generates the chart: 
<script>
    scp = new RGraph.SVG.SemiCircularProgress({
        id: 'chart-container',
        min: 0,
        max: 100,
        value: 78,
        options: {
            title: 'A meter depicting the tanks volume',
            colors: ['Gradient(#060:#3c3:#3c3:#3c3)'],
            backgroundColor: 'rgba(0,255,0,0.25)',
            labelsCenterDecimals: 1,
            labelsCenterUnitsPost: '%'
        }
    }).draw();
</script>
